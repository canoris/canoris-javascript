 32  	/**
 33  	 * Method to create uri for request. 
 34  	 * @param {string} uri Contains the standerd string for request. 
 35  	 * @param {array} args Contains the arguments that needs to be placed in standard string.
 36  	 * @returns {string} uri Contains the total uri for the request.
 37  	 */
 38 	this._uri = function(uri, args){
 39 		for (a in args) {
 40 			uri = uri.replace(/<[\w_]+>/, args[a]);
 41 		};
 42 		uri = CAN.CanorisData.getBaseUri() + uri;
 43 		return uri;
 44 	};
 45 	
 46 	/**
 47 	 * Method to split reference uri in uri and paramters
 48 	 * @param {string} uri Contains the uri to split in uri and parameters 
 49 	 */
 50 	this._splitUri = function(uri)
 51 	{
 52 		//test uri for ? -> contains parameters
 53 		if(/\?/.test(uri)){
 54 			//split up uri
 55 			var splittedUri = uri.split(/\?/);
 56 			//save baseUri
 57 			var baseUri = splittedUri[0];
 58 			//save parameters
 59 			var parameters = splittedUri[1];
 60 			//test params for & -> contains more then one parameter
 61 			if(/\&/.test(parameters)){	
 62 				//split parameters
 63 				var params =  parameters.split(/\&/); 
 64 			}
 65 			//else safe single parameter in array to be able to create parameter object
 66 			else{var params = [parameters]; };
 67 			//create object to place parameters in
 68 			var parametersObject = {}; 
 69 			//split up parameters strings and place in object
 70 			for( a in params)
 71 			{	
 72 				var param = params[a].split(/\=/);
 73 				eval("parametersObject."+ param[0] + "=\"" + param[1] + "\"");
 74 			}
 75 			//return array with baseUri and paramtersObject
 76 			return [baseUri, parametersObject];
 77 		} else{
 78 			return [uri, false];	
 79 		}
 80 	}
 81 }

 95 	/**
 96 	 * create the logger and appenders and set these to the passed levels
 97 	 * @param {string} cLogLevel Contains the level for the console logger.
 98 	 * @param {string} sLogLevel Contains the level for the server logger.
 99 	 */
100 	CAN.Log.setLogger = function(cLogLevel, sLogLevel){
101 		CAN.Log.log = log4javascript.getLogger("logger");
102 		if(CAN.Log.checkLevel(cLogLevel)){
103 			var popUpAppender = new log4javascript.PopUpAppender();
104 			popUpAppender.setThreshold(eval("log4javascript.Level." + cLogLevel));
105 			CAN.Log.log.addAppender(popUpAppender);
106 		}
107 		if(CAN.Log.checkLevel(sLogLevel)){
108 			var ajaxAppender = new log4javascript.AjaxAppender(CAN._LOGADDRESS);
109 			ajaxAppender.setThreshold( eval("log4javascript.Level." + sLogLevel));
110 			CAN.Log.log.addAppender(ajaxAppender);
111 		}
112 	}
113 	
114 	/**
115 	 * check level method is used to check if the passed level is available
116 	 * @param {string} level
117 	 */
118 	CAN.Log.checkLevel = function(level){
119 		if( level == "ALL" 	|| level == "TRACE" || level == "DEBUG" || level == "INFO" || 
120 			level == "WARN" || level == "ERROR"	|| level == "FATAL"	|| level == "OFF")
121 		{	return level; }
122 		else {return false}
123 	}					
124 	
125 	/**
126 	 * Method to be able to show object in logger 
127 	 * @param {Object} obj Contains the object that needs to be shown in logger
128 	 * @param {level} level Contains the level, used for spaceing when object contains objects
129 	 */
130 	CAN.Log.showObject = function(obj, level){
131 		var str = '';
132 		if(!level){ var level = 0;};
133 	    var space = "\n";
134 		for(var i = 1 ; i <= level; i++){ space +="	";}  
135 		for(property in obj){
136 	    	//append property to string
137 	        str += space + property + " = " + obj[property];
138 			 // and we are inside the required number of levels
139 	        if((typeof(obj[property]) == 'object') && (obj[property] != null) && property != 'object')
140 	        { str += CAN.Log.showObject(obj[property], level + 1);} 
141 			else if((typeof(obj[property]) == 'object') && (obj[property] != null) && property == 'object')
142 			{ str += " --> Contains itself."}
143 			 
144 	    }
145 	    return str;
146 	}

149 		if(address){CAN._LOGADDRESS = address;}

240 /**
241  * Standard error function, used whith json request when no error callback method is passed.
242  * @static
243  * @param {object} XMLHttpRequest
244  * @param {string} textStatus Describes the type of error that occurred.
245  * @param {object} errorThrown Optional exception object.
246  */
247 CAN.RequestCreator.standardErrorMethod = function(XMLHttpRequest, textStatus, errorThrown){	
248 	CAN.Log.log.error( "Request failed, request = \n" + CAN.Log.showObject(XMLHttpRequest) + 
249 						"\nerror of Request =  " + textStatus +	"\nthrown error =  " + errorThrown); 
250 	CAN.Log.log.error( "Request failed, request = \n" + CAN.Log.showObject(XMLHttpRequest) +

252 	throw new Error("RequestCreator error " + textStatus);
253 };

255 /**
256  * Create a request to server. 
257  * @static
258  * @param {string} uri Holds the uri for the request.
259  * @param {function} succesCallback Will be called when the request succeeds.
260  * @param {fucntion} errorCallback Will be called when the request fails
261  * (no XMLHttpRequest and specified error passed to errorCallback with use of jsonp)
262  * @param {array} params Contains the parameters as String that need to be send with the request.
263  */
264 CAN.RequestCreator.createGetReq = function(uri, succesCallback, errorCallback, params){	
265 	CAN.Log.log.debug(	"inside RequestCreator.createGetReq, passed args:" +
266 							"\n__uri: " + uri + 
267 							"\n__succesCallback is: " + succesCallback + 
268 							"\n__errorCallback is: " + errorCallback + 
269 		 					"\n__parameters: " + CAN.Log.showObject(params));
270 	//get api key, needed to create a request	
271 	var aKey = CAN.CanorisData.getApiKey();
272 	//create parameter object, with api key and passed paramameters
273 	var dataParams = {api_key: aKey};
274 	//if params are passed, extend dataParams with params
275 	if (params) { $.extend(dataParams, params); };
276 	//check if errorCallback is a function, 
277 	//if not -> use standard error method, else combine standard errorCallback with passed errorcallback
278 	if(!$.isFunction(errorCallback)){	
279 		var errorCallback = CAN.RequestCreator.standardErrorMethod;
280 	} else {
281 		var errorCallback = function(XMLHttpRequest, textStatus, errorThrown){
282 			CAN.RequestCreator.standardErrorMethod(XMLHttpRequest, textStatus, errorThrown);
283 			errorCallback(XMLHttpRequest, textStatus, errorThrown);
284 		};
285 	}; 
286 	//check with wich type of json the request needs to be made
287 	if(CAN.RequestCreator.useJson){
288 		//send a json XMLHttpRequest 
289 		$.ajax({
290 			url: uri,
291 			dataType: 'json',
292 			data: dataParams,
293 			success: succesCallback,
294 			error: errorCallback,
295 			type: 'GET'
296 		});
297 		
298 	} else{
299 		//send a jsonp request
300 		$.jsonp({
301 			url: uri + "?callback=?",
302 			data: dataParams,
303 			timeout: CAN.RequestCreator.timeout,
304 			success: succesCallback,
305 			error: errorCallback
306 		});
307 	}
308 }

328 /**
329  * Get item of CanorisObject
330  * @param {string} item Contains the item name.
331  * @returns {object/string} item 
332  */
333 CAN.CanorisObject.prototype.getItem = function(item){
334 	//check if CanorisObject object properties are loaded
335 	if(this.loaded){
336 		var myString =  "this.properties." + item;
337 		//check if propertie asked for exists
338 		if (eval(myString)) {
339 			return eval(myString);
340 		}
341 		//propertie doesn't exist, throw error
342 		else{throw new SyntaxError( item + " does not exist in " + this)}
343 	}
344 	//properties aren't loaded, throw error
345 	else {throw new Error("Properties of " + this + " are not loaded.")};
346 }

356 /**
357  * Reload the CanorisObject.
358  * @param {function} succesCallback Will be called when the request succeeds.
359  * @param {fucntion} errorCallback Will be called when the request fails (no XMLHttpRequest passed and specified error with use of jsonp).
360  */
361 CAN.CanorisObject.prototype.update = function(succesCallback, errorCallback){
362 	if(this.properties.ref)
363 	{
364 		//split uri into uri and parameters object
365 		var uriAndParams = CAN._splitUri(this.properties.ref);
366 		//save this (file, collection, pager, that is calling the update method) inside var object	
367 		var object = this;
368 		//create clossure around object, with which properties can be set 
369 		var setProps = new function(props){
370 		return function(props){ object.properties = props;}
371 		}
372 		//call createGetReq method, use above setProps method in callback method to be able to save properties
373 		CAN.RequestCreator.createGetReq(uriAndParams[0], function(inProperties){
374 			setProps(inProperties);
375 			if($.isFunction(succesCallback)){succesCallback(inProperties); };
376 		}, errorCallback, uriAndParams[1]);	
377 		this.properties = false; 	
378 	}else{ throw new Error("properties of object does not exist, unable to update because no ref. is present");};
379 }

762 /**
763  * @class Canoris class is the API of the Canoris Javascript Library
764  * @constructor
765  * @param {String} aKey Contains the api key that is going to be used.
766  * @param {Boolean} useJson Contains a boolean indicating the use of Json, otherwise Jsonp
767  * @param {String} levelConsoleLog Contains the level of the consoleLogger available levels are: 
768  * ALL TRACE DEBUG INFO WARN ERROR FATAL OFF
769  * @param {String} levelServerLog Contains the level of the serverLogger, available levels see levelDebugLog
770  * @param {String} serverLogAddress Contains the addres to which the serverLogger makes calls
771  * @returns {Object} { pagers, getFilesPager, getCollectionsPager, getCollectionFilesPager, files, 
772  * 			getFile, collections, getCollection, text2phonemes, setTimeout }. 
773  * <br/>See <a href="canorisLib_howto.html">How-to</a> on how to use canoris object.
774  */
775 Canoris = function(aKey, useJson, levelConsoleLog, levelServerLog, serverLogAddress){
776 //
777 //\npagers -> contains the pagers created with getFilesPager, getCollectionsPager, getCollectionFilesPager methods
778 	
779 	//check if there already exist a object of Canoris class (singleton)
780 	if(!Canoris.object)
781 	{
782 		if (aKey) {
783 			
784 			//set use of json, logAddress and loggers levels
785 			CAN.RequestCreator.setUseJson(useJson);
786 			CAN.Log.setLogAddress(serverLogAddress);	
787 			CAN.Log.setLogger(levelConsoleLog, levelServerLog);
788 					
789 			//empty pagers object, will be ued to hold pagers
790 			var pagers = {};
791 			//savePager function to be able to store pagers(by using a closure for pagers variable)
792 			var savePager = function(name){
793 				return function(val){
794 					pagers[name] = val;
795 				}
796 			};
797 			//getFilesPager function to call CAN.Pager.getFilesPager with pagers clossure function
798 			var getFilesPager = function(namePager, pageNr, succesCallback, errorCallback)
799 			{	CAN.Pager.getFilesPager(savePager(namePager), pageNr, succesCallback, errorCallback); };
800 			//getCollectionsPager function to call CAN.Pager.getCollectionsPager with pagers clossure function
801 			var getCollectionsPager = function(namePager, pageNr, succesCallback, errorCallback)
802 			{	CAN.Pager.getCollectionsPager(savePager(namePager), pageNr, succesCallback, errorCallback); };
803 			//getCollectionFilesPager function to call CAN.Pager.getCollectionFilesPager with pagers clossure function
804 			var getCollectionFilesPager = function(cKey, namePager, pageNr, succesCallback, errorCallback)
805 			{	CAN.Pager.getCollectionFilesPager(cKey, savePager(namePager), pageNr, succesCallback, errorCallback); };
806 						
807 			//empty files object, will be used to hold files 
808 			var files = {};
809 			//saveFile function to be able to store files (by using a closure for files variable) 
810 			var saveFile = function(name){
811 				return function(val){
812 					files[name] = val;
813 				}
814 			};
815 			//getFile function to call CAN.File.getFile with files clossure function
816 			var getFile = function(fKey, nameFile, succesCallback, errorCallback)
817 			{	CAN.File.getFile(fKey, saveFile(nameFile), succesCallback, errorCallback); };
818 			
819 			//empty collections object, will be used to hold collections 
820 			var collections = {};
821 			//saveCollection function to be able to store collections (by using a closure for collection variable) 
822 			var saveCollection = function(name){
823 				return function(val){
824 					collections[name] = val;
825 				}
826 			};
827 			//getCollection function to call CAN.Collection.getCollection with collections clossure function
828 			var getCollection = function(cKey, nameCollection, succesCallback, errorCallback)
829 			{	CAN.Collection.getCollection(cKey, saveCollection(nameCollection), succesCallback, errorCallback); };
830 				
831 			//set api key
832 			CAN.CanorisData.setApiKey(aKey); 
833 			 			
834 			//create Canoris object, containing public variables and methods of Canoris Library						
835 			var object = {
836 				
837 				pagers: pagers,
838 				getFilesPager:  getFilesPager, 
839 				getCollectionsPager: getCollectionsPager,
840 				getCollectionFilesPager: getCollectionFilesPager,
841 				
842 				files: files,
843 				getFile: getFile, 
844 
845 				collections: collections,
846 				getCollection: getCollection,
847 								
848 				text2phonemes: CAN.Text2Phonemes.translate,
849 				
850 				setTimeout: CAN.RequestCreator.setTimeout
851 				
852 //change this in a reset method -> all files change like this right? ASK VINCENT what he wants!
853 			};
854 		}
855 		//no correct api key passed -> throw error
856 		else {throw "you didn't pass a api_key to constructor canoris"; }
857 	}
858 	
859 	//return object of Canoris class
860 
861 	return object;	
862 }
